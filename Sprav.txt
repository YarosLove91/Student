// Начальная инициализация
// Директивы:
//--local — для данного проекта.
//--global — для данного пользователя.
//--system — для данного компьютера (в рамках ОС).
// Присваиваем имя и почту
git config --global user.name "Имя"
git config --global user.email "ваша@почта.net"

// Создание репозитория через консоль
// Создание папки через консоль
mkdir имя_папки 
// Переход в папку
cd имя_папки  
// Создаем новый репозитории
git init

// Изменение проекта
// Просмотр статуса репозитория
git status
// Добавление измененных файликов в индекс
git add имена_файликов
// Создаем новую ревизию
// Рекомендую делать через специальную команду -m, а то придется
// Работать в "родном редакторе"
git commit -m "Комментарий"



// Удаление из индекса
git rm имя_файла
// Просмотр истории
git log
git log --graph

// Показать все ветки в репозитарии
git branch
// Создать ветку
git branch имя 

//Переключиться на ветку (все должно быть сохранено !)
checkout имя 

// Создать ветку и переключиться на неё
git checkout -b имя

// Слияние веток
git checkout x1 
git merge x2

// Удаление ветки
git branch –d имя_ветки

// Редактор "Notepad++"
git config --global core.editor "\"c:\Program Files\Notepad++\notepad++.exe\" %*"

// Редактор "Блокнот"
git config --global core.editor "\"c:\Windows\System32\notepad.exe\" %*"
